Poniżej jest przykładowy skrypt w Pythonie, który używa biblioteki `selenium` do automatyzacji przeglądarki Chrome i pobiera hasła z formularza logowania. Musisz zainstalować bibliotekę `selenium` przed uruchomieniem skryptu.

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.common.exceptions import TimeoutException

# Ustawienia przeglądarki
options = webdriver.ChromeOptions()
options.add_argument('headless')  # Uruchom przeglądarkę w trybie headless
options.add_argument('disable-gpu')  # Wyłącz akcelerator GPU
options.add_argument('window-size=1920,1080')  # Ustaw rozmiar okna

# Utwórz obiekt przeglądarki
driver = webdriver.Chrome(options=options)

# Przejdź do strony logowania
driver.get('https://example.com/logowanie')

# Pobierz element formularza logowania
try:
    form = WebDriverWait(driver, 10).until(
        EC.presence_of_element_located((By.ID, 'form_logowanie'))
    )
except TimeoutException:
    print('Nie udało się znaleźć formularza logowania')
    driver.quit()
    exit()

# Pobierz element pola hasła
try:
    password_input = WebDriverWait(driver, 10).until(
        EC.presence_of_element_located((By.ID, 'password'))
    )
except TimeoutException:
    print('Nie udało się znaleźć pola hasła')
    driver.quit()
    exit()

# Pobierz wartość pola hasła
password = password_input.get_attribute('value')

# Zamknij przeglądarkę
driver.quit()

# Wyświetl wartość hasła
print(password)
```
